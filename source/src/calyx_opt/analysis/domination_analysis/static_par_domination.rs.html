10/// Computes Dominators Across Static Pars.
11/// Assumes each control stmt has already been given the appropraite IDs.
12pub struct StaticParDomination {
13    /// (nodes for static control can only be enables or if stmts... we don't suppport invokes yet)
14    /// maps par ids -> (map of node ids -> (first interval for which node is live, relative to parent par))
15    pub node_timing_map: HashMap<u64, HashMap<u64, (u64, u64)>>,
16    /// maps par ids -> (map of node ids -> (first interval for which node is live, relative to parent par)), but these enables *may* execute
17    pub node_maybe_timing_map: HashMap<u64, HashMap<u64, (u64, u64)>>,
18    /// name of component
19    component_name: ir::Id,
20}

22impl Debug for StaticParDomination {
23    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
24        writeln!(
25            f,
26            "This maps ids of par blocks to \"node timing maps\", which map node ids to the first interval (i,j) that the node (i.e., enable/invoke/if conditional) is active for, \n relative to the start of the given par block"
27        )?;
28        write!(
29            f,
30            "============ Map for Component \"{}\"",
31            self.component_name
32        )?;
33        writeln!(f, " ============")?;
34        let node_must_map = self.node_timing_map.clone();
35        let node_may_map = self.node_maybe_timing_map.clone();
36        // get all par ids and iterate thru them. Sort for deterministic ordering
37        let must_par_ids: HashSet<&u64> = node_must_map.keys().collect();
38        let may_par_ids: HashSet<&u64> = node_may_map.keys().collect();
39        let mut par_ids: Vec<_> = must_par_ids.union(&may_par_ids).collect();
40        par_ids.sort();
41        for par_id in par_ids.into_iter() {
42            write!(f, "========")?;
43            write!(f, "Par Node ID: {par_id:?}")?;
44            writeln!(f, "========")?;
45            write!(f, "====")?;
46            write!(f, "MUST EXECUTE")?;
47            writeln!(f, "====")?;
48            // print the "must executes" for the given par id
49            match node_must_map.get(par_id) {
50                None => (),
51                Some(map) => {
52                    let mut vec1: Vec<_> = map.iter().collect();
53                    // sort vec1 to get deterministic ordering
54                    vec1.sort_by(|(k1, _), (k2, _)| k1.cmp(k2));
55                    for (enable_id, interval) in vec1 {
56                        // print enable id -- (interval)
57                        write!(f, "{enable_id:?} -- ")?;
58                        writeln!(f, "{interval:?}")?;
59                    }
60                }
61            }
62            write!(f, "====")?;
63            write!(f, "MAY EXECUTE")?;
64            writeln!(f, "====")?;
65            // print the "may executes" for the given par id
66            match node_may_map.get(par_id) {
67                None => (),
68                Some(map) => {
69                    let mut vec1: Vec<_> = map.iter().collect();
70                    // sort vec1 to get deterministic ordering
71                    vec1.sort_by(|(k1, _), (k2, _)| k1.cmp(k2));
72                    for (enable_id, interval) in vec1 {
73                        // print enable id -- (interval)
74                        write!(f, "{enable_id:?} -- ")?;
75                        writeln!(f, "{interval:?}")?;
76                    }
77                }
78            }
79        }
80        writeln!(f)
81    }
82}

84impl StaticParDomination {
85    /// Construct a live range analysis.
86    pub fn new(control: &mut ir::Control, comp_name: ir::Id) -> Self {
87        let mut timing_map = StaticParDomination {
88            component_name: comp_name,
89            ..Default::default()
90        };
91
92        timing_map.build_time_map(control);
93
94        timing_map
95    }

97    /// returns a HashMap that maps node ids -> set of nodes that dominate it
98    /// It will only return the node ids that are dominated within the same
99    /// static par block, not all the dominators for the entire control program
100    pub fn get_static_dominators(&mut self) -> HashMap<u64, HashSet<u64>> {
101        let mut static_dom_map: HashMap<u64, HashSet<u64>> = HashMap::new();
102
103        for (par_id, node_interval_mapping) in &self.node_timing_map {
104            let empty_map = HashMap::new();
105            // these are the nodes that *may* execute for the given par id
106            let may_execute_enables =
107                match self.node_maybe_timing_map.get(par_id) {
108                    Some(mapping) => mapping,
109                    None => &empty_map,
110                };
111            // Very simple/naive algorithm
112            // for each "must" execute nodes, it can either dominate:
113            // one of the "may" execute nodes, or dominate another "must" execute node
114            // "may" execute nodes cannot dominate anybody
115            for (node_id1, (_, end1)) in node_interval_mapping {
116                // check if node_id1 dominates any of the "may" execute nodes
117                for (may_enable, (may_beg, _)) in may_execute_enables {
118                    if end1 <= may_beg {
119                        static_dom_map
120                            .entry(*may_enable)
121                            .or_default()
122                            .insert(*node_id1);
123                    }
124                }
125                // check if node_id1 dominates any of the other "must" execute nodes
126                for (enable_id2, (beg2, _)) in node_interval_mapping {
127                    if end1 <= beg2 {
128                        static_dom_map
129                            .entry(*enable_id2)
130                            .or_default()
131                            .insert(*node_id1);
132                    }
133                }
134            }
135        }
136        static_dom_map
137    }

139    // updates self.node_timing_map if guaranteed_execution is true, otherwise
140    // updates self.node_maybe_timing_map.
141    // Also returns the "state = (par_id, cur_clock)" after the invoke/enable has occured
142    // assumes that there is a cur_state = (par_id, cur_clock)
143    // also, id is the id of the node, and latency is the latency of the node
144    fn update_node(
145        &mut self,
146        id: u64,
147        latency: u64,
148        cur_state: (u64, u64),
149        guaranteed_execute: bool,
150    ) -> (u64, u64) {
151        let (par_id, cur_clock) = cur_state;
152        // if we are guaranteed execution, then can update self.node_timing_map
153        // otherwise we must updateself.node_maybe_timing_map
154        let enable_mappings = if guaranteed_execute {
155            self.node_timing_map.entry(par_id).or_default()
156        } else {
157            self.node_maybe_timing_map.entry(par_id).or_default()
158        };
159        // we already have recorded an earlier execution of the node, so we don't care about a later execution
160        if enable_mappings.get(&id).is_none() {
161            enable_mappings.insert(id, (cur_clock, cur_clock + latency));
162        }
163        (par_id, cur_clock + latency)
164    }
165
166    // Recursively updates self.enable_timing_map
167    // This is a helper function for fn `build_time_map`.
168    // Read comment for that function to see what this function is doing
169    // returns the resulting "state"
170    fn build_time_map_static(
171        &mut self,
172        sc: &ir::StaticControl,
173        // cur_state = Some(parent_par_id, cur_clock) if we're inside a static par, None otherwise.
174        // parent_par_id = Node ID of the static par that we're analyzing
175        // cur_clock = current clock cycles we're at relative to the start of parent_par
176        cur_state: Option<(u64, u64)>,
177        // whether sc is guaranteed to execute (i.e., not in an `if` statement branch)
178        guaranteed_execution: bool,
179    ) -> Option<(u64, u64)> {
180        match sc {
181            ir::StaticControl::Empty(_) => cur_state,
182            ir::StaticControl::Enable(ir::StaticEnable { group, .. }) => {
183                if let Some(cur_state_unwrapped) = cur_state {
184                    let enable_id = ControlId::get_guaranteed_id_static(sc);
185                    let latency = group.borrow().get_latency();
186                    Some(self.update_node(
187                        enable_id,
188                        latency,
189                        cur_state_unwrapped,
190                        guaranteed_execution,
191                    ))
192                } else {
193                    cur_state
194                }
195            }
196            ir::StaticControl::Invoke(inv) => {
197                if let Some(cur_state_unwrapped) = cur_state {
198                    let invoke_id = ControlId::get_guaranteed_id_static(sc);
199                    let latency = inv.latency;
200                    Some(self.update_node(
201                        invoke_id,
202                        latency,
203                        cur_state_unwrapped,
204                        guaranteed_execution,
205                    ))
206                } else {
207                    cur_state
208                }
209            }
210            ir::StaticControl::If(ir::StaticIf {
211                tbranch, fbranch, ..
212            }) => {
213                // should look thru if branches to see if they have static pars
214                // inside of them, but static if branches don't help us with
215                // dominators across pars (since we're not sure if they execute),
216                // so we can't add any enables within the if branches
217                self.build_time_map_static(tbranch, cur_state, false);
218                self.build_time_map_static(fbranch, cur_state, false);
219                let if_id =
220                    ControlId::get_guaranteed_attribute_static(sc, BEGIN_ID);
221                if let Some(cur_state_unwrapped) = cur_state {
222                    self.update_node(
223                        if_id,
224                        1,
225                        cur_state_unwrapped,
226                        guaranteed_execution,
227                    );
228                }
229
230                cur_state.map(|(parent_par, cur_clock)| {
231                    (parent_par, cur_clock + sc.get_latency())
232                })
233            }
234            ir::StaticControl::Repeat(ir::StaticRepeat { body, .. }) => {
235                // we only need to look thru the body once either way, since we only
236                // care about the *first* execution of a node
237                self.build_time_map_static(
238                    body,
239                    cur_state,
240                    guaranteed_execution,
241                );
242                cur_state.map(|(par_id, cur_clock_cycle)| {
243                    (par_id, cur_clock_cycle + sc.get_latency())
244                })
245            }
246            ir::StaticControl::Seq(ir::StaticSeq { stmts, .. }) => {
247                // this works whether or not cur_state is None or Some
248                let mut new_state = cur_state;
249                for stmt in stmts {
250                    new_state = self.build_time_map_static(
251                        stmt,
252                        new_state,
253                        guaranteed_execution,
254                    );
255                }
256                new_state
257            }
258            ir::StaticControl::Par(ir::StaticPar { stmts, .. }) => {
259                // We know that all children must be static
260                // Analyze the Current Par
261                for stmt in stmts {
262                    self.build_time_map_static(
263                        stmt,
264                        Some((ControlId::get_guaranteed_id_static(sc), 0)),
265                        true,
266                    );
267                }
268                // If we have nested pars, want to get the clock cycles relative
269                // to the start of both the current par and the nested par.
270                // So we have the following code to possibly get the clock cycles
271                // relative to the parent par.
272                match cur_state {
273                    Some((cur_parent_par, cur_clock)) => {
274                        for stmt in stmts {
275                            self.build_time_map_static(
276                                stmt,
277                                cur_state,
278                                guaranteed_execution,
279                            );
280                        }
281                        Some((cur_parent_par, cur_clock + sc.get_latency()))
282                    }
283                    None => None,
284                }
285            }
286        }
287    }

289    // Recursively updates self.node_timing_map and self.node_maybe_timing_map
290    // Takes in Control block `c`
291    // they both map maps par ids -> (maps of node ids -> (first interval for which the node is live))
292    fn build_time_map(&mut self, c: &ir::Control) {
293        match c {
294            ir::Control::Invoke(_)
295            | ir::Control::Empty(_)
296            | ir::Control::Enable(_) => (),
297            ir::Control::Par(ir::Par { stmts, .. })
298            | ir::Control::Seq(ir::Seq { stmts, .. }) => {
299                for stmt in stmts {
300                    self.build_time_map(stmt)
301                }
302            }
303            ir::Control::If(ir::If {
304                tbranch, fbranch, ..
305            }) => {
306                self.build_time_map(tbranch);
307                self.build_time_map(fbranch);
308            }
309            ir::Control::While(ir::While { body, .. })
310            | ir::Control::Repeat(ir::Repeat { body, .. }) => {
311                self.build_time_map(body);
312            }
313            ir::Control::Static(sc) => {
314                self.build_time_map_static(sc, None, true);
315            }
316            ir::Control::FSMEnable(_) => {
317                todo!()
318            }
319        }
320    }
